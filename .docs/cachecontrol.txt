const http = require('http');
http.createServer(function (req, res) {
 // Set the response headers to control caching behavior
 res.writeHead(200, {
 'Content-Type': 'text/html',
 'Cache-Control': 'no-cache, no-store, must-revalidate',
 'Pragma': 'no-cache',
 'Expires': '0'
 });
// Write the response
 res.end('Hello World!');
}).listen(8080);
